Inheritance

Do

Don't

Agenda

  • Put all logic into the parent component
  • Extend a child from the parent and adjust the specific logic
  • Inherit templates from the parent; overwrite where necessary

Modal Component

<div id="{{this.id}}">
  Modal opened: {{this.opened}}

  {{#if (has-block-params)}}
    {{yield (hash
      header=(component 'header' modal=this)
      body=(component 'body' modal=this)
    )}}
  {{else if (has-block)}}
    <Body>{{yield}}</Body>
  {{else}}
    <Header></Header>

    <Body></Body>
  {{/if}}
</div>

import SparklesComponent from 'sparkles-component';

export interface ModalArgs {
  open?: boolean;
}

export default class ModalInheritanceComponent<T extends ModalArgs> extends SparklesComponent<T> {
}

<div>
  ModalInheritanceHeader
  {{#if (has-block)}}
    {{yield}}
  {{else}}
    Modal Header
  {{/if}}
</div>

<div>
  {{#if (has-block)}}
    {{yield}}
  {{else}}
    Modal Body
  {{/if}}
</div>

Modal Dialog Component

// @ts-ignore
export { default } from '../modal/template';

import ModalInheritanceComponent, { ModalArgs } from '../modal/component';
import { arg } from 'sparkles-decorators';

interface ModalDialogArgs extends ModalArgs {
  title?: string;
}

export default class ModalDialogInheritanceComponent extends ModalInheritanceComponent<ModalDialogArgs> {
  @arg title!: string;
}

<div style="color: red">
  {{#if (has-block)}}
    {{yield}}
  {{else}}
    {{@modal.title}}
  {{/if}}
</div>
